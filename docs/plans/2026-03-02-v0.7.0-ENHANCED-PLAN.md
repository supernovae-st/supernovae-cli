```rust
// src/ai/recommendations.rs
use rig::providers::claude;

pub async fn recommend_packages(project_path: &Path) -> Result<Vec<Recommendation>> {
    // 1. Analyze project
    let context = analyze_project(project_path).await?;

    // 2. Build prompt
    let prompt = format!(
        "Project type: {}\nDependencies: {:?}\nStructure: {:?}\n\
        Suggest 3-5 relevant SuperNovae packages (workflows, agents, skills).",
        context.project_type,
        context.dependencies,
        context.structure
    );

    // 3. Call Claude
    let client = claude::ClientBuilder::new()
        .api_key(get_anthropic_key()?)
        .build();

    let response = client
        .agent("claude-sonnet-4.5")
        .preamble("You are a package recommendation expert.")
        .prompt(&prompt)
        .temperature(0.7)
        .chat()
        .await?;

    // 4. Parse response
    parse_recommendations(&response)
}

#[derive(Debug)]
struct ProjectContext {
    project_type: String,      // "rust-web", "typescript-api", etc.
    dependencies: Vec<String>,  // Cargo.toml, package.json
    structure: Vec<String>,     // src/, docs/, tests/
}

async fn analyze_project(path: &Path) -> Result<ProjectContext> {
    let mut context = ProjectContext {
        project_type: String::new(),
        dependencies: Vec::new(),
        structure: Vec::new(),
    };

    // Detect project type
    if path.join("Cargo.toml").exists() {
        context.project_type = "rust".to_string();
        context.dependencies = parse_cargo_toml(path)?;
    } else if path.join("package.json").exists() {
        context.project_type = "typescript".to_string();
        context.dependencies = parse_package_json(path)?;
    }

    // Analyze structure
    context.structure = std::fs::read_dir(path)?
        .filter_map(|entry| entry.ok())
        .filter(|entry| entry.file_type().ok()?.is_dir())
        .map(|entry| entry.file_name().to_string_lossy().to_string())
        .collect();

    Ok(context)
}
```

```rust
// src/commands/mcp_playground.rs
use dialoguer::{Select, Input};
use serde_json::Value;

pub async fn run() -> Result<()> {
    println!("🎮 MCP Tool Playground\n");

    // 1. List available MCP servers
    let servers = list_mcp_servers()?;

    let server_names: Vec<String> = servers.iter()
        .map(|s| format!("{} ({})", s.name, s.tools.len()))
        .collect();

    let server_idx = Select::new()
        .with_prompt("Select MCP server")
        .items(&server_names)
        .interact()?;

    let selected_server = &servers[server_idx];

    // 2. List tools in server
    let tool_names: Vec<String> = selected_server.tools.iter()
        .map(|t| format!("{} - {}", t.name, t.description))
        .collect();

    let tool_idx = Select::new()
        .with_prompt("Select tool")
        .items(&tool_names)
        .interact()?;

    let selected_tool = &selected_server.tools[tool_idx];

    // 3. Interactive parameter input
    println!("\n📝 Enter parameters for {}:", selected_tool.name);

    let mut params = serde_json::Map::new();

    for param in &selected_tool.parameters {
        let value: String = Input::new()
            .with_prompt(&format!("{} ({})", param.name, param.type_))
            .interact_text()?;

        params.insert(param.name.clone(), Value::String(value));
    }

    // 4. Execute tool
    println!("\n🚀 Executing {}...\n", selected_tool.name);

    let result = execute_mcp_tool(
        &selected_server.command,
        &selected_tool.name,
        &Value::Object(params)
    ).await?;

    // 5. Display result
    println!("📊 Result:\n");
    println!("{}", serde_json::to_string_pretty(&result)?);

    // 6. Save as workflow?
    if Confirm::new()
        .with_prompt("\n💾 Save as Nika workflow?")
        .interact()?
    {
        save_as_workflow(&selected_tool, &params)?;
        println!("✓ Saved to workflow.nika.yaml");
    }

    Ok(())
}

async fn execute_mcp_tool(
    server_command: &str,
    tool_name: &str,
    params: &Value
) -> Result<Value> {
    // Start MCP server
    let mut server = Command::new("sh")
        .arg("-c")
        .arg(server_command)
        .stdin(Stdio::piped())
        .stdout(Stdio::piped())
        .spawn()?;

    // Send tool call via stdio
    let request = json!({
        "jsonrpc": "2.0",
        "id": 1,
        "method": "tools/call",
        "params": {
            "name": tool_name,
            "arguments": params
        }
    });

    let stdin = server.stdin.as_mut().unwrap();
    serde_json::to_writer(stdin, &request)?;

    // Read response
    let stdout = server.stdout.as_mut().unwrap();
    let response: Value = serde_json::from_reader(stdout)?;

    Ok(response)
}

fn save_as_workflow(tool: &McpTool, params: &Map<String, Value>) -> Result<()> {
    let workflow = format!(
        r#"schema: nika/workflow@0.9
workflow: test-{tool}
description: Test workflow for {tool}

mcp:
  servers:
    server:
      command: "{command}"

tasks:
  - id: test_tool
    invoke: {tool}
    params:
{params}
"#,
        tool = tool.name,
        command = "...", // actual command
        params = serde_json::to_string_pretty(&params)?
    );

    std::fs::write("workflow.nika.yaml", workflow)?;
    Ok(())
}
```

```rust
// src/soul/mod.rs
use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize)]
pub struct Soul {
    pub identity: Identity,
    pub state: State,
    pub memory: Vec<MemoryEntry>,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct Identity {
    pub name: String,
    pub personality: String,
    pub memory_retention: Duration,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct State {
    pub last_topics: Vec<String>,
    pub learned_preferences: HashMap<String, String>,
    pub cache_strategy: String,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct MemoryEntry {
    pub timestamp: DateTime<Utc>,
    pub interaction_type: String,  // "query", "correction", "feedback"
    pub content: String,
    pub metadata: HashMap<String, Value>,
}

impl Soul {
    /// Load soul from package
    pub fn load(package_name: &str) -> Result<Self> {
        let soul_path = get_package_path(package_name)?.join("soul.yaml");

        if !soul_path.exists() {
            return Ok(Self::default());
        }

        let yaml = std::fs::read_to_string(soul_path)?;
        let soul: Soul = serde_yaml::from_str(&yaml)?;

        // Load persistent memory
        let memory_path = get_soul_memory_path(package_name)?;
        if memory_path.exists() {
            let memory_json = std::fs::read_to_string(memory_path)?;
            let mut soul = soul;
            soul.memory = serde_json::from_str(&memory_json)?;
            Ok(soul)
        } else {
            Ok(soul)
        }
    }

    /// Save soul state
    pub fn save(&self, package_name: &str) -> Result<()> {
        let memory_path = get_soul_memory_path(package_name)?;

        // Prune old memories based on retention policy
        let retention_cutoff = Utc::now() - self.identity.memory_retention;
        let active_memory: Vec<_> = self.memory.iter()
            .filter(|entry| entry.timestamp > retention_cutoff)
            .collect();

        let memory_json = serde_json::to_string_pretty(&active_memory)?;
        std::fs::write(memory_path, memory_json)?;

        Ok(())
    }

    /// Add new memory entry
    pub fn remember(&mut self, interaction_type: &str, content: String, metadata: HashMap<String, Value>) {
        self.memory.push(MemoryEntry {
            timestamp: Utc::now(),
            interaction_type: interaction_type.to_string(),
            content,
            metadata,
        });
    }

    /// Query memories by type or time range
    pub fn recall(&self, filter: MemoryFilter) -> Vec<&MemoryEntry> {
        self.memory.iter()
            .filter(|entry| filter.matches(entry))
            .collect()
    }
}

fn get_soul_memory_path(package_name: &str) -> Result<PathBuf> {
    let home = dirs::home_dir().ok_or_else(|| SpnError::HomeNotFound)?;
    let soul_dir = home.join(".spn/souls");
    std::fs::create_dir_all(&soul_dir)?;

    Ok(soul_dir.join(format!("{}.json", package_name)))
}
```

**What:** Seamlessly search and install from public + private registries

```rust
// src/index/multi_registry.rs
use futures::stream::{self, StreamExt};

pub async fn search_all_registries(query: &str) -> Result<Vec<PackageInfo>> {
    let config = Config::load()?;

    // Search all registries in parallel
    let futures: Vec<_> = config.registries.iter()
        .map(|registry| search_single_registry(registry, query))
        .collect();

    let results = stream::iter(futures)
        .buffer_unordered(3)  // Max 3 concurrent searches
        .collect::<Vec<_>>()
        .await;

    // Merge results with conflict resolution
    let mut all_packages = Vec::new();
    let mut seen = HashSet::new();

    for result in results {
        match result {
            Ok(packages) => {
                for pkg in packages {
                    // Deduplicate by name (prefer private registry)
                    if !seen.contains(&pkg.name) {
                        seen.insert(pkg.name.clone());
                        all_packages.push(pkg);
                    } else if pkg.is_private {
                        // Replace public with private version
                        all_packages.retain(|p| p.name != pkg.name);
                        all_packages.push(pkg);
                    }
                }
            }
            Err(e) => {
                eprintln!("⚠️  Registry search failed: {}", e);
            }
        }
    }

    // Sort by relevance + source priority
    all_packages.sort_by(|a, b| {
        // Private packages first
        match (a.is_private, b.is_private) {
            (true, false) => std::cmp::Ordering::Less,
            (false, true) => std::cmp::Ordering::Greater,
            _ => {
                // Then by downloads
                b.downloads.cmp(&a.downloads)
            }
        }
    });

    Ok(all_packages)
}

async fn search_single_registry(
    registry: &Registry,
    query: &str
) -> Result<Vec<PackageInfo>> {
    match registry.auth {
        RegistryAuth::Public => {
            search_public_registry(&registry.url, query).await
        }
        RegistryAuth::GitHub => {
            let token = get_github_token()?;
            search_github_registry(&registry.url, query, &token).await
        }
        RegistryAuth::Bearer(ref token) => {
            search_authenticated_registry(&registry.url, query, token).await
        }
    }
}
```
